### 应用资源配置清单
```shell script
kubectl apply -f http://k8s-yaml.qytang.com/emptydir/nginx-curl-emptydir.yaml
```

### pod描述(主要用于Pod内多容器间交换文件)
```shell script
kubectl describe pod nginx-curl-emptydir-fc977bd5c-jjzxc

Volumes:
  emptydir:
    Type:       EmptyDir (a temporary directory that shares a pod's lifetime)
```

### filebeat采集日志实例
```shell script
    spec:
      containers:
      - name: nameko-nginx
        image: harbor.qytang.com/public/nginx:log_local
        ports:
        - containerPort: 80
          protocol: TCP
        imagePullPolicy: Always
        volumeMounts:
          - name: static-images
            mountPath: /static
          - name: configmap-volume
            mountPath: /etc/nginx/conf.d  
          - name: logdir
            mountPath: /var/log/nginx
      - name: filebeat
        image: harbor.qytang.com/public/filebeat:configured
        imagePullPolicy: Always
        env:
        volumeMounts:
        - mountPath: /logdir        
          name: logdir
      volumes:
        - name: configmap-volume
          configMap:
            name: nameko-nginx-cm
        - name: static-images
          nfs:
           server: harbor
           path: /data/nfs-volume/nameko_nginx/static 
        - emptyDir: {}
          name: logdir
```
